def create
  @product = Product.new(params[:product])
  if @product.save
    redirect_to product_path(@product)
  else
    render :new
  end
end

def update
  @product.update_attributes(params[:product])
  @product.save
  if @product.retired
    @product.line_items.map(&:destroy)
  end
  redirect_to products_path
end
