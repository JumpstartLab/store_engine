def create
  cart_id = current_cart.id
  user = login(params[:email], params[:password])
  if user
    session[:cart_id] = cart_id
    redirect_back_or_to root_url, :notice => "Logged in!"
  else
    flash.now.alert = "Email or password was invalid"
    render :new
  end
end
